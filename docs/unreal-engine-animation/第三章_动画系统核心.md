## 3.2 动画混合技术

### 3.2.1 线性混合算法

**基础混合实现**：
```cpp
// 两个动画的线性混合
FTransform BlendTwoAnimations(const FTransform& TransformA, 
                             const FTransform& TransformB, 
                             float BlendAlpha) {
    FTransform Result;
    
    // 位置混合
    Result.SetTranslation(
        FMath::Lerp(TransformA.GetTranslation(), 
                   TransformB.GetTranslation(), 
                   BlendAlpha)
    );
    
    // 旋转混合（四元数球面插值）
    Result.SetRotation(
        FQuat::Slerp(TransformA.GetRotation(), 
                    TransformB.GetRotation(), 
                    BlendAlpha)
    );
    
    // 缩放混合
    Result.SetScale3D(
        FMath::Lerp(TransformA.GetScale3D(), 
                   TransformB.GetScale3D(), 
                   BlendAlpha)
    );
    
    return Result;
}
```

**多动画混合策略**：
```cpp
// 多个动画的加权混合
void BlendMultipleAnimations(const TArray<FTransform>& Transforms,
                            const TArray<float>& Weights,
                            FTransform& OutResult) {
    if (Transforms.Num() == 0) return;
    
    // 归一化权重
    float TotalWeight = 0.0f;
    for (float Weight : Weights) {
        TotalWeight += Weight;
    }
    
    // 加权混合
    FVector BlendedTranslation = FVector::ZeroVector;
    FQuat BlendedRotation = FQuat::Identity;
    FVector BlendedScale = FVector::OneVector;
    
    for (int32 i = 0; i < Transforms.Num(); ++i) {
        float NormalizedWeight = Weights[i] / TotalWeight;
        
        BlendedTranslation += Transforms[i].GetTranslation() * NormalizedWeight;
        BlendedScale += Transforms[i].GetScale3D() * NormalizedWeight;
        
        // 旋转需要特殊处理
        if (i == 0) {
            BlendedRotation = Transforms[i].GetRotation();
        } else {
            BlendedRotation = FQuat::Slerp(BlendedRotation, 
                                          Transforms[i].GetRotation(), 
                                          NormalizedWeight);
        }
    }
    
    OutResult.SetTranslation(BlendedTranslation);
    OutResult.SetRotation(BlendedRotation);
    OutResult.SetScale3D(BlendedScale);
}
```

### 3.2.2 加法混合技术

**加法动画原理**：
```cpp
// 加法混合实现
FTransform AdditiveBlend(const FTransform& BasePose,
                        const FTransform& AdditivePose,
                        float AdditiveWeight) {
    FTransform Result = BasePose;
    
    // 位置加法
    Result.AddToTranslation(AdditivePose.GetTranslation() * AdditiveWeight);
    
    // 旋转加法（四元数乘法）
    FQuat AdditiveRotation = AdditivePose.GetRotation() * AdditiveWeight;
    Result.SetRotation(AdditiveRotation * BasePose.GetRotation());
    
    return Result;
}
```

**应用场景示例**：
- **呼吸动画**：在基础姿势上叠加呼吸动作
- **表情动画**：在面部基础表情上叠加情绪变化
- **武器瞄准**：在站立姿势上叠加瞄准动作

### 3.2.3 骨骼分层混合

**分层混合策略**：
```cpp
// 骨骼分层混合配置
struct FBoneBlendProfile {
    TMap<FName, float> BoneWeights; // 骨骼名称 -> 混合权重
    
    float GetBoneWeight(FName BoneName) const {
        return BoneWeights.FindRef(BoneName);
    }
};

// 分层混合实现
void LayeredBlendPerBone(const FTransform& BasePose,
                        const FTransform& BlendPose,
                        const FBoneBlendProfile& Profile,
                        FTransform& OutResult) {
    
    for (int32 BoneIndex = 0; BoneIndex < Skeleton->GetNumBones(); ++BoneIndex) {
        FName BoneName = Skeleton->GetBoneName(BoneIndex);
        float Weight = Profile.GetBoneWeight(BoneName);
        
        if (Weight > 0.0f) {
            // 对该骨骼进行混合
            FTransform BlendedTransform = BlendTwoAnimations(
                BasePose.GetBoneTransform(BoneIndex),
                BlendPose.GetBoneTransform(BoneIndex),
                Weight
            );
            OutResult.SetBoneTransform(BoneIndex, BlendedTransform);
        } else {
            // 保持基础姿势
            OutResult.SetBoneTransform(BoneIndex, BasePose.GetBoneTransform(BoneIndex));
        }
    }
}
```

**伤害区域通知**：
```cpp
// 伤害区域激活通知
UCLASS()
class UAnimNotifyState_DamageZone : public UAnimNotifyState {
    GENERATED_BODY()
    
private:
    TArray<AActor*> DamagedActors; // 已伤害的演员列表
    
public:
    virtual void NotifyBegin(USkeletalMeshComponent* MeshComp, UAnimSequenceBase* Animation, float TotalDuration) override {
        DamagedActors.Empty();
    }
    
    virtual void NotifyTick(USkeletalMeshComponent* MeshComp, UAnimSequenceBase* Animation, float FrameDeltaTime) override {
        // 检测碰撞并应用伤害
        TArray<FOverlapResult> Overlaps;
        FCollisionQueryParams QueryParams;
        
        if (MeshComp->GetWorld()->OverlapMultiByChannel(
            Overlaps,
            MeshComp->GetSocketLocation("WeaponTip"),
            FQuat::Identity,
            ECC_GameTraceChannel1, // 伤害检测通道
            FCollisionShape::MakeSphere(50.0f),
            QueryParams
        )) {
            for (const FOverlapResult& Overlap : Overlaps) {
                if (AActor* HitActor = Overlap.GetActor()) {
                    if (!DamagedActors.Contains(HitActor)) {
                        ApplyDamage(HitActor);
                        DamagedActors.Add(HitActor);
                    }
                }
            }
        }
    }
    
    virtual void NotifyEnd(USkeletalMeshComponent* MeshComp, UAnimSequenceBase* Animation) override {
        DamagedActors.Empty();
    }
};
```

### 3.5.1 动画数据缓存

**变换缓存系统**：
```cpp
// 动画变换缓存
class FAnimationTransformCache {
private:
    TMap<float, TArray<FTransform>> CachedTransforms; // 时间 -> 骨骼变换
    
public:
    bool GetCachedTransforms(float Time, TArray<FTransform>& OutTransforms) {
        if (CachedTransforms.Contains(Time)) {
            OutTransforms = CachedTransforms[Time];
            return true;
        }
        return false;
    }
    
    void CacheTransforms(float Time, const TArray<FTransform>& Transforms) {
        CachedTransforms.Add(Time, Transforms);
    }
};
```
